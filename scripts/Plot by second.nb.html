Plot by second

requirements

require(ggplot2)

Load

sos-only <- read.csv("../data/sos-only.csv")

Preview

Head

count.by.second$Created.Date <- as.POSIXct(strptime(count.by.second$Created.Date, format = "%m/%d/%Y %I:%M:%S %p")) 
head(count.by.second)

Aggregate

count.by.second$Month <- as.POSIXct(cut(count.by.second$Created.Date,
  breaks = "month"))
count.by.second$Week <- as.POSIXct(cut(count.by.second$Created.Date,
  breaks = "week",
  start.on.monday = FALSE))

Plot

ggplot(count.by.second, aes(Week, n)) + 
  scale_x_datetime() + xlab("") + ylab("Week")+
  stat_summary(fun.y = sum, # adds up all observations for the week
    geom = "line")
